symbol = input("Choose a operator: +, -, * or /: ")
number = int(input("What number would you like? "))
number2 = int(input("What number would you like? "))

if symbol == "+":
              print(number + number2)
if symbol == "/":
              print(number / number2)
if symbol == "*":
              print(number * number2)
if symbol == "-":
              print(number - number2)
